#nullable enable
using static Seekatar.Mask;
using System.Linq;
using System;
namespace Seekatar.OptionToStringGenerator
{
    /// <summary>
    /// Class for OptionsToString extension methods.
    /// </summary>
    public static partial class ClassExtensions
    {
        /// <summary>
        /// Converts the options to a string representation.
        /// </summary>
        /// <param name="o">object</param>
        /// <param name="extraIndent">any extra indent</param>
        /// <param name="titleSuffix">suffix for the title of the object</param>
        /// <returns>string representation of values, redacted as configured</returns>
        public static string OptionsToString(this Test.FormatOptionsJson? o, string extraIndent = "", string titleSuffix = "")
        {
            return $@"{{
  ""Test.FormatOptionsJson"" : {{
{extraIndent}  ""Secrets""  : {Format(o?.Secrets,asJson:true,formatMethod:(o) => Test.FormatOptionsJson.MyFormatter(o),noQuote:false)}
  }}
}}";
        }
    }
}
