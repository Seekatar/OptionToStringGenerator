#nullable enable
namespace Seekatar.OptionToStringGenerator
{
    public static partial class PropertyExtensions
    {
        internal static string OptionsToString(this Test.MyRecord o)
        {
            return $@"Test.MyRecord:
  Number   : {OptionsToStringAttribute.Format(o.Number)}
  Name     : {OptionsToStringAttribute.Format(o.Name,prefixLen:0,suffixLen:0)}
  SerialNo : {OptionsToStringAttribute.Format(o.SerialNo,prefixLen:0,suffixLen:3)}
";
        }
    }
}
